          --health-cmd pg_isready
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5

          - 5432:5432
    
    steps:
      - name: ğŸ¦• Checkout code

          pip install -r requirements.txt
          pip install pytest pytest-cov pytest-asyncio httpx
      
      - name: ğŸ§ª Run tests
        working-directory: ./backend
        env:
          DATABASE_URL: postgresql://testuser:testpass@localhost:5432/dinoreserve_test

      - name: ğŸ“¥ Install dependencies
        working-directory: ./frontend

      - name: ğŸ” Lint code

      - name: ğŸ—ï¸ Build application

      - name: ğŸ“¤ Upload build artifacts

    name: Backend Code Quality
    runs-on: ubuntu-latest
    
    steps:
      - name: ğŸ¦• Checkout code
        uses: actions/checkout@v3
      
      - name: ğŸ Set up Python

          python-version: '3.11'
      
      - name: ğŸ“¦ Install linting tools

      - name: ğŸ¨ Check code formatting with black

      - name: ğŸ¦• Checkout code

          cache-to: type=gha,mode=max
      
      - name: ğŸ—ï¸ Build frontend image

          tags: dinoreserve-frontend:test

          cache-to: type=gha,mode=max
